8    def build_equivariant_nn(L=5, w=2):
9        """Creates a permutation-equivariant neural network. 
10    
11       Args: 
12       L (int): Number of equivariant layers. 
13       w (int): Number of filters in the convolutional layers. 
14    
15       Returns: 
16       model (tf.keras.Model): Compiled equivariant neural network. 
17       """
18       model = Sequential()
19   
20       # Equivariant affine transformations (Conv1D layers)
21       for i in range(L - 1):
22           model.add(
23               Conv1D(filters=w, kernel_size=1, activation='relu', padding='same', input_shape=(m, d) if i == 0 else None))
24   
25       # Final equivariant affine transformation (Output layer)
26       model.add(Conv1D(filters=1, kernel_size=1, activation=None, padding='same'))
27   
28       # Flatten the output to match expected one-hot output shape
29       model.add(Flatten())
30   
31       # Compile the model
32       model.compile(optimizer=Adam(learning_rate, epsilon),
33                     loss='mse', metrics=['mse'])
34   
35       return model
